  <action>
Replace the entire content of `components/body_camera.gd` with the implementation below.

Design decisions baked in:
- `_physics_process` instead of `_process` — syncs with physics body velocity
- `zoom_default = 0.25` — slightly more zoomed in than before at rest (felt better)
- `zoom_min = 0.10` — maximum zoom-out at high sustained speed
- `speed_threshold = 600.0` — below this, zoom-out does not begin (ignores jitter)
- `zoom_out_onset_time = 0.6` — ship must be above threshold for this many seconds before
  zoom-out starts (the "sustained fast movement" delay)
- `zoom_in_delay = 0.8` — after slowing down, camera waits this long before zooming back in
- `zoom_out_speed = 1.5` — lerp weight per second for zooming out (feels responsive)
- `zoom_in_speed = 0.8` — lerp weight per second for zooming in (slower = more cinematic)
- Acceleration bonus: `accel_bonus` scales zoom-out lerp speed by how hard the ship is
  accelerating (harder accel = faster zoom-out response), capped at 3x
- The unused `zoom_levels` export is removed to clean up dead code

```gdscript
class_name BodyCamera extends Camera2D

@export var body: Body

# Zoom bounds — lower value = more zoomed out in Godot 2D
const ZOOM_DEFAULT  := 0.25   # zoom when stopped / slow
const ZOOM_MIN      := 0.10   # maximum zoom-out at high sustained speed

# Speed at which zoom-out begins (px/s). Below this = treated as stopped.
const SPEED_THRESHOLD := 600.0

# How long (s) the ship must exceed SPEED_THRESHOLD before zoom-out starts.
const ZOOM_OUT_ONSET := 0.6

# How long (s) after slowing down before zoom-in begins.
const ZOOM_IN_DELAY := 0.8

# Lerp weights per second (applied as lerp(current, target, weight * delta)).
const ZOOM_OUT_RATE := 1.5
const ZOOM_IN_RATE  := 0.8

# Max speed reference used to derive target zoom (must be > SPEED_THRESHOLD).
const SPEED_MAX := 4000.0

# Internal state
var _prev_speed    := 0.0
var _time_fast     := 0.0   # seconds above SPEED_THRESHOLD
var _time_slow     := 0.0   # seconds below SPEED_THRESHOLD after being fast
var _zooming_out   := false  # true once onset delay satisfied
var _zoom_current  := ZOOM_DEFAULT


func _ready() -> void:
	_zoom_current = ZOOM_DEFAULT
	zoom = Vector2(_zoom_current, _zoom_current)


func _physics_process(delta: float) -> void:
	if not body:
		return

	# --- Position: follow body directly (no smoothing needed, body has physics) ---
	global_position = body.global_position

	# --- Speed and acceleration ---
	var speed := body.linear_velocity.length()
	var accel := abs(speed - _prev_speed) / delta  # px/s²
	_prev_speed = speed

	# --- State machine: fast / slow timers ---
	if speed > SPEED_THRESHOLD:
		_time_fast += delta
		_time_slow  = 0.0
	else:
		_time_slow += delta
		_time_fast  = 0.0

	# Onset: start zooming out once ship has been fast long enough
	if not _zooming_out and _time_fast >= ZOOM_OUT_ONSET:
		_zooming_out = true

	# Reset: stop zooming out (begin zoom-in delay) once ship slows down
	if _zooming_out and speed <= SPEED_THRESHOLD and _time_slow >= ZOOM_IN_DELAY:
		_zooming_out = false

	# --- Target zoom ---
	var zoom_target: float
	if _zooming_out:
		# Map speed to zoom range. Higher speed = more zoomed out (lower value).
		var ratio := clampf((speed - SPEED_THRESHOLD) / (SPEED_MAX - SPEED_THRESHOLD), 0.0, 1.0)
		zoom_target = lerpf(ZOOM_DEFAULT, ZOOM_MIN, ratio)
	else:
		zoom_target = ZOOM_DEFAULT

	# --- Lerp zoom toward target ---
	var lerp_rate: float
	if zoom_target < _zoom_current:
		# Zooming out — acceleration bonus makes it feel more responsive
		var accel_bonus := clampf(accel / 2000.0, 0.0, 2.0)  # 0..2 bonus multiplier
		lerp_rate = ZOOM_OUT_RATE * (1.0 + accel_bonus)
	else:
		lerp_rate = ZOOM_IN_RATE

	_zoom_current = lerpf(_zoom_current, zoom_target, clampf(lerp_rate * delta, 0.0, 1.0))
	zoom = Vector2(_zoom_current, _zoom_current)
```
