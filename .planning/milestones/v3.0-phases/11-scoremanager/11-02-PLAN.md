Create `components/score-manager.gd` with the following complete implementation. This file is loaded as an autoload singleton (registered in project.godot by Plan 01).

```gdscript
class_name ScoreManager
extends Node

## Signals for Phase 12 HUD consumption
signal score_changed(new_score: int, delta: int)
signal multiplier_changed(new_multiplier: int)
signal combo_updated(combo_count: int)
signal combo_expired(bonus_awarded: int)

## Constants (per D-06, D-10, D-11)
const MULTIPLIER_CAP: int = 16
const COMBO_TIMEOUT: float = 5.0
const COMBO_BONUS_PER_KILL: int = 25

## State
var total_score: int = 0
var kill_count: int = 0
var wave_multiplier: int = 1
var combo_count: int = 0

## Internal references
var _player: Node = null
var _combo_timer: Timer = null
var _combo_audio: AudioStreamPlayer = null


func _ready() -> void:
    # Combo timer — one-shot, 5 second timeout (per D-10)
    # Follows Timer.new() pattern from MountableWeapon (shot_timer, reload_timer)
    _combo_timer = Timer.new()
    _combo_timer.wait_time = COMBO_TIMEOUT
    _combo_timer.one_shot = true
    _combo_timer.timeout.connect(_on_combo_expired)
    add_child(_combo_timer)

    # Combo audio — non-positional AudioStreamPlayer (per D-14)
    _combo_audio = AudioStreamPlayer.new()
    _combo_audio.stream = preload("res://sounds/combo.wav")
    add_child(_combo_audio)

    # Deferred player lookup — autoloads run before scene nodes are ready
    # Matches WaveManager._find_player() deferred pattern (Pitfall 4)
    call_deferred("_find_player")


func _find_player() -> void:
    _player = get_tree().get_first_node_in_group("player")
    if not _player:
        push_warning("[ScoreManager] No node in group 'player' found")
        return
    # Connect to player health_changed for multiplier reset (per D-07)
    _player.health_changed.connect(_on_player_health_changed)
    print("[ScoreManager] Connected to player health_changed signal")


## Called by world.gd to wire WaveManager signals (per D-08)
func connect_to_wave_manager(wm: Node) -> void:
    wm.wave_completed.connect(_on_wave_completed)
    print("[ScoreManager] Connected to WaveManager wave_completed signal")


## Called by WaveManager._spawn_enemy() for each spawned enemy
## Uses .bind(enemy) to pass the enemy reference to the handler
func register_enemy(enemy: Node) -> void:
    if enemy.has_signal("died"):
        enemy.died.connect(_on_enemy_died.bind(enemy))


# --- Kill Scoring (per D-04, D-05, SCR-04) ---

func _on_enemy_died(enemy: Node) -> void:
    var base_score: int = enemy.score_value if "score_value" in enemy else 0
    var kill_score: int = base_score * wave_multiplier

    kill_count += 1
    _increment_combo()

    total_score += kill_score
    score_changed.emit(total_score, kill_score)

    # Enemy type name from class_name for print output
    var enemy_type: String = enemy.get_script().get_global_name() if enemy.get_script() else "Unknown"
    print("[ScoreManager] Kill: %s +%d (x%d) = %d | total: %d" % [
        enemy_type, base_score, wave_multiplier, kill_score, total_score
    ])


# --- Combo Chain (per D-09, D-10, D-11, D-12, D-13) ---

func _increment_combo() -> void:
    if combo_count == 0:
        # First kill in potential chain — start timer, no audio yet (per D-09)
        combo_count = 1
        _combo_timer.start()
        return

    # Subsequent kills — increment combo, restart timer, play audio
    combo_count += 1
    _combo_timer.start()
    _play_combo_sound(combo_count)
    combo_updated.emit(combo_count)
    print("[ScoreManager] Combo x%d" % combo_count)


func _on_combo_expired() -> void:
    if combo_count < 2:
        # Single kill, no real combo — just reset
        combo_count = 0
        return

    # Award combo bonus (per D-11): combo_count * 25 * wave_multiplier
    var bonus: int = combo_count * COMBO_BONUS_PER_KILL * wave_multiplier
    total_score += bonus
    score_changed.emit(total_score, bonus)
    combo_expired.emit(bonus)
    print("[ScoreManager] Combo x%d expires, bonus +%d | total: %d" % [combo_count, bonus, total_score])
    combo_count = 0
    combo_updated.emit(0)


func _play_combo_sound(current_combo: int) -> void:
    # Semitone pitch steps (per D-13):
    # combo_count 2 = pitch 1.0 (base), 3 = 1.0595, 4 = 1.12, etc.
    # Formula: pitch_scale = pow(1.0595, combo_count - 1)
    # But since combo_count 2 should be base pitch (1.0), we use combo_count - 2
    # because pow(1.0595, 0) = 1.0 at combo_count 2
    _combo_audio.pitch_scale = pow(1.0595, current_combo - 2)
    _combo_audio.play()


# --- Wave Multiplier (per D-06, D-07, D-08) ---

func _on_wave_completed(_wave_number: int) -> void:
    # Double multiplier on wave completion, cap at 16 (per D-06)
    if wave_multiplier < MULTIPLIER_CAP:
        wave_multiplier = mini(wave_multiplier * 2, MULTIPLIER_CAP)
    multiplier_changed.emit(wave_multiplier)
    print("[ScoreManager] Wave complete, multiplier x%d" % wave_multiplier)


func _on_player_health_changed(old_health: int, new_health: int) -> void:
    # Reset multiplier on damage only (per D-07)
    # health_changed only fires when health decreases (Body.damage guard)
    # but double-check here for safety
    if new_health < old_health and wave_multiplier > 1:
        wave_multiplier = 1
        multiplier_changed.emit(wave_multiplier)
        print("[ScoreManager] Damage taken, multiplier reset to x1")
```

IMPORTANT pitch clarification per D-13: "Combo kill 2 = 1.0 (base pitch), kill 3 = 1.0595, kill 4 = 1.12". This means:
- combo_count=2: `pow(1.0595, 2-2) = pow(1.0595, 0) = 1.0` (correct)
- combo_count=3: `pow(1.0595, 3-2) = pow(1.0595, 1) = 1.0595` (correct)
- combo_count=4: `pow(1.0595, 4-2) = pow(1.0595, 2) = 1.1225` (correct)

The exponent is `combo_count - 2`, NOT `combo_count - 1` as stated in D-13's formula. D-13 says "pitch_scale = pow(1.0595, combo_count - 1)" but then specifies "Combo kill 2 = 1.0 (base pitch)" which requires exponent 0 at combo_count=2. The concrete examples take priority over the formula. Use `combo_count - 2`.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Enemy death -> ScoreManager | Score calculation uses enemy's score_value property; tamperable in local files |
| Wave completion -> Multiplier | Wave-completed signal drives multiplier doubling; no external input |
| Combo timer -> Bonus award | Timer-based combo expiry awards bonus score; timer is engine-managed |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-04 | Tampering | score_value in enemy .tscn | accept | Offline single-player game. Players can edit .tscn files locally, but there is no competitive multiplayer or server-side validation to protect. |
| T-11-05 | Tampering | total_score in memory | accept | GDScript vars are in-process memory. Memory editing tools (Cheat Engine) could modify total_score. No server authority exists. Acceptable for single-player. |
| T-11-06 | Information Disclosure | Print output of scores | accept | Debug print statements expose scoring internals to console. This is intentional for Phase 11 verification. Phase 12+ may optionally remove prints. |
| T-11-07 | Elevation of Privilege | ScoreManager autoload global access | accept | Any script can call ScoreManager methods. This is by design (autoload pattern). No sensitive operations beyond score tracking. |
| T-11-08 | Denial of Service | Rapid combo_count increment (integer overflow) | mitigate | combo_count is `int` (64-bit in GDScript). Even at 1000 kills/second for 1 hour = 3.6M, well within int range. pitch_scale = pow(1.0595, N) would produce inaudible frequencies at very high N, but Timer expiry at 5s naturally caps real combo chains to ~50 kills. No explicit cap needed. |
</threat_model>
